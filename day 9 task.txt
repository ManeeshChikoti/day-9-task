     if(xhr.status>=200 && xhr.status<300)
  {
     let data=JSON.parse(this.response);
     data=data.filter((ref)=>
     { 
         if((ref.continents).toString()==="Asia")
         { 
             console.log(ref.name.common);
         }
    })
  }
}
xhr.send();

c.Print the following details name, capital, flag using forEach function

Answer:

var xhr=new XMLHttpRequest();
xhr.open('GET',"https://restcountries.com/v3.1/all");
xhr.onload=function(){
     if(xhr.status>=200 && xhr.status<300)
  {
     let data=JSON.parse(this.response);
     //flag is used to get code of the country flag in restcountries
     //if we want print the flag image we cane use flags instead of flag
     //i tried to print flag and flags
      data=data.forEach((element) => {
         console.log(  `countryname:${element.name.common},
          capital:${element.capital},
           flagcode:${element.flag},
         flag:${element.flags.png}`
         )
     });
    
  }
}
xhr.send();

d.Print the total population of countries using reduce function

Answer:
var xhr=new XMLHttpRequest();
xhr.open('GET',"https://restcountries.com/v3.1/all");
xhr.onload=function(){
     if(xhr.status>=200 && xhr.status<300)
  {
     let data=JSON.parse(this.response);
     let Totalpopulation=  data.reduce((acc,cur)=>
     {
         return acc+cur.population;
      },0)
      console.log(`total population is ${Totalpopulation}`);
}
}
xhr.send();
